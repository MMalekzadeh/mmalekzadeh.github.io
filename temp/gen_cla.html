import numpy as np

## Information of all data subjects
dsinfs = dsbldr.get_ds_infos()
print("\nData Subjects Info is Created: shape = ",dsinfs.shape)

females = (dsinfs[dsinfs[:,4]==0])[:,0]
males =   (dsinfs[dsinfs[:,4]==1])[:,0]
print("--> Females: ", list(females.astype('int')))
print("--> Males: ", list(males.astype('int')))
num_folds = 4
label = "wlk"


losses = np.zeros(num_folds)
accuracies = np.zeros(num_folds)
cf_rprt = {}
cf_cfmt = {}

for fold in range(num_folds):
    print("\n---->> ## Fold ",str(fold+1),":")
    train_list = []
    validate_list = []
    test_list = []
    
    for j in range(len(females)):
        if j%num_folds == fold:
            test_list.append(int(females[j]))
            continue
        train_list.append(int(females[j]))    
    
    for j in range(len(males)):
        if j%num_folds == fold:
            test_list.append(int(males[j]))
            continue
        train_list.append(int(males[j]))    
    
    validate_list.append(train_list[fold+1])
    validate_list.append(train_list[fold+6])
    validate_list.append(train_list[fold+12])
    train_list.remove(train_list[fold+1])  
    train_list.remove(train_list[fold+5]) 
    train_list.remove(train_list[fold+10])      
    
    print("Train List is = ",train_list)
    print("Validate List is = ",validate_list)
    print("Test List is = ",test_list)
    
    
    ## existing labels: "dwn", "up", "sit", "std", "wlk", "jog"  
    train_one_by_gender = dsbldr.all_one_by_gender(dsinfs, label, train_list)
    print("shape of '",label,"' acitviy Training dataset: shape = ",train_one_by_gender.shape)
    
    validate_one_by_gender = dsbldr.all_one_by_gender(dsinfs, label, validate_list)
    print("shape of '",label,"' acitviy Validation dataset: shape = ",validate_one_by_gender.shape)
    
    test_one_by_gender = dsbldr.all_one_by_gender(dsinfs, label, test_list)
    print("shape of '",label,"' acitviy Test dataset: shape = ",test_one_by_gender.shape)
        
        
            
    ## Train Model on Similar
    print("_____________________________________________")
    print("Learning: Gender Classification")
    train_data, train_labels , mean, std = ts2secs.time_series_to_section(train_one_by_gender,
                                                                         standardize = True)

    validate_data, validate_labels, mean, std = ts2secs.time_series_to_section(validate_one_by_gender,
                                                                               standardize = True,
                                                                               mean = mean, std = std)
    
    test_data, test_labels, mean, std = ts2secs.time_series_to_section(test_one_by_gender,
                                                                       standardize = True,
                                                                       mean = mean, std = std)
    
#    index = np.array(range(0,len(train_data),20))
#    val_d = train_data[index]
#    val_l = train_labels[index]
#    new_train_data = np.delete(train_data,index, axis=0)
#    new_train_labels = np.delete(train_labels,index, axis=0)
            
    typeof= "gen_cla"
    val_loss = cnn_act_rec.train(typeof,
                      train_data,
                      train_labels,
                      test_data,
                      test_labels,
                      validate_data,
                      validate_labels,
                      epoc=50,
                      vbs=2,
                      validate= True)
    
    val_loss = np.array(val_loss)
    val_loss[:4] = max(val_loss) 
    epoc = (val_loss.argmin())+1    

    f_loss, f_accuracy, f_cf_cfmt, f_cf_rprt = cnn_act_rec.train(typeof,
                                                                 train_data,
                                                                 train_labels,
                                                                 test_data,
                                                                 test_labels,
                                                                 validate_data,
                                                                 validate_labels,
                                                                 epoc=epoc,
                                                                 vbs=0,
                                                                 validate= False
                                                                 )
    
    print("_____________________________________________")
    ## Save results of current fold 
    losses[fold] = f_loss
    accuracies[fold] = f_accuracy
    cf_cfmt[fold] = f_cf_cfmt 
    cf_rprt[fold] = f_cf_rprt
    
print("\n***~~~ Final Results:")
plotcfr.plot(typeof, losses, accuracies, cf_cfmt, cf_rprt, num_folds)
